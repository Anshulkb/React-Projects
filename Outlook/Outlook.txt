import MessageBox from 'sap/m/MessageBox';
import { API_ENDPOINTS } from './endpoints';
import { CodeResponse, ErrorMsg, TokenData, TokenResponse } from './schemas';
import Log from 'sap/base/Log';

const url = 'https://graph.microsoft.com/v1.0/me/messages';
let token = '',
  tokenExpire = 0,
  tokenRecievedAt = 0,
  refreshToken = '',
  mailBody = '',
  mailSubject = '';

export function openLoginPopup(): Promise<string> {
  return new Promise((resolve, reject) => {
    const client = crypto.randomUUID();

    const loginWindow = window.open(
      API_ENDPOINTS.exportOutlook.outlook_redirect(client),
      'popupWindow',
      'width=600,height=400'
    );

    if (!loginWindow) {
      reject(new Error('Failed to open login popup'));
      return;
    }

    const socket = new WebSocket(`wss://${window.location.host}/ws/${client}`);

    socket.onerror = () => {
      reject(new Error('Unable to get login details. Please try again.'));
    };
    socket.onmessage = (event: CodeResponse) => {
      const data = JSON.parse(event.data) as TokenResponse;
      if (data.status === 'ready') {
        token = data.token.access_token;
        tokenExpire = data.token.expires_in;
        tokenRecievedAt = Date.now();
        refreshToken = data.token.refresh_token;
        socket.close();
        if (loginWindow) {
          loginWindow.close();
        }
        resolve(data.token.access_token);
      }
    };
  });
}

export async function generateMail(mailContent: string) {
  mailBody = mailContent;
  checkSubjectPresent();
  try {
    if (tokenExpire === 0 && tokenRecievedAt === 0) {
      await openLoginPopup();
      openComposedMail();
    } else if (isTokenExpired(tokenExpire, tokenRecievedAt)) {
      getRefreshToken()
        .then((data) => {
          if (Object.keys(data).length != 0) {
            token = data.access_token;
            tokenExpire = data.expires_in;
            tokenRecievedAt = Date.now();
            refreshToken = data.refresh_token;
            openComposedMail();
          } else {
            throw new Error('Unable to fetch token. Please login again.');
          }
        })
        .catch((err: ErrorMsg) => {
          MessageBox.error(err?.message);
        });
    } else {
      await loggingCDI();
      openComposedMail();
    }
  } catch (err) {
    throw new Error(String(err));
  }
}
async function getRefreshToken(): Promise<TokenData> {
  const res = await fetch(API_ENDPOINTS.exportOutlook.get_refreshToken(refreshToken));
  if (!res.ok) {
    throw new Error('An unexpected error occurred while fetching session details.');
  }

  const data = (await res.json()) as TokenData;
  if (data && data.error != undefined) {
    throw new Error('Unable to fetch token from the server. Please refresh the page and try again');
  }
  return data;
}

async function createDraftEmail(accessToken: string) {
  mailBody = decodeHTMLEntities(mailBody);
  if (!checkHTMLTags(mailBody)) {
    mailBody = marked.parse(mailBody);
  }
  const cleanBody = DOMPurify.sanitize(mailBody);
  const email = {
    subject: mailSubject,
    body: {
      contentType: 'HTML',
      content: cleanBody
    },
    toRecipients: [
      {
        emailAddress: {
          address: ''
        }
      }
    ]
  };

  const response = await fetch(url, {
    method: 'POST',
    headers: {
      Authorization: `Bearer ${accessToken}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(email)
  });

  if (!response.ok) {
    throw new Error('An error occured while composing mail');
  }

  const draftMessage = (await response.json()) as { id: string };
  return draftMessage.id;
}
function isTokenExpired(expiresIn: number, tokenReceivedTime: number) {
  const currentTime = Date.now();
  const expiryTime = tokenReceivedTime + expiresIn * 1000;

  return currentTime >= expiryTime;
}
function openComposedMail() {
  createDraftEmail(token)
    .then((response) => {
      const messageId = encodeURIComponent(response);
      const outlookWebUrl = `https://outlook.office365.com/mail/deeplink/compose/${messageId}`;

      window.open(outlookWebUrl, '_blank');
    })
    .catch((err) => {
      if (err instanceof Error) {
        MessageBox.error(err?.message);
      } else {
        Log.error('Error occured in compose mail.');
      }
    });
}

function checkSubjectPresent() {
  const regex = /^(?:subj(?:ect)?\s*\d*|subj(?:ect)?\s*[A-Z])\s*:\s*(.*)$/im;
  const match = regex.exec(mailBody);
  mailSubject = match ? match[1].trim() : '';
  if (mailSubject.length > 0) {
    mailSubject = mailSubject.replace(/(\*\*\*|\*\*|\*|`)/g, '');
    mailSubject = mailSubject.replace(/\[([^\]]+)\]\([^)]+\)/g, '$1');
  }
}

function decodeHTMLEntities(text: string) {
  const txt = document.createElement('textarea');
  txt.innerHTML = text;
  return txt.value;
}

function checkHTMLTags(text: string) {
  const htmlTagRegex = /<\/?[a-z][\s\S]*>/i;
  return htmlTagRegex.test(text);
}

async function loggingCDI() {
  await fetch(API_ENDPOINTS.exportOutlook.log_outlook_cdi(0.01))
    .then((res) => {
      if (!res.ok) {
        throw new Error('An unexpected error occurred while logging export data to outlook.');
      }
    })
    .catch((err) => {
      Log.error(String(err));
    });
}
